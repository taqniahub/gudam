####################################
'Ways to Return values from Function
'####################################
 
'a. Single return value
 
Function Temp()
    Dim x
    X=5
     
    Temp= x
End Function
    
'b.Multiple values return:
Concatenate variables
Function Temp()
    Dim x,y
    X=5
    Y=6
     
    Temp= x & "~"y
End Function
 
'c. Save Variables in array
Function Temp()
    Dim x,y, aResult(1)
    x=5
    y=6
    aResult (0) = x
    aResult(1) = y
    Temp=aResult
End Function
 
'd. Return Object
Set ret = Temp()
X = ret.Item(1)
Y= ret.item(2)
